代码示例 — 过滤算子

from ops.base_op import BaseOp

class LengthFilterOp(BaseOp):
    """过滤文本长度低于阈值的文档"""
    
    def execute(self, data: list[dict]) -> list[dict]:
        min_length = self.config.get("min_length", 100)
        return [
            d for d in data 
            if len(d.get("text", "")) >= min_length
        ]
继承 BaseOp 实现 execute config 获取参数

代码示例 — LLM算子

from ops.base_op import BaseOp
from ops.llms import LLMClient

class QualityEvaluatorOp(BaseOp):
    """使用LLM评估文档质量"""
    
    def execute(self, data: list[dict]) -> list[dict]:
        client = LLMClient(self.config.get("model"))
        results = []
        for doc in data:
            score = client.evaluate(doc["text"])
            doc["quality_score"] = score
            results.append(doc)
        return results
LLMClient集成 批量处理 新增字段输出
